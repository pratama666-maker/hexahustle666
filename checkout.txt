  <!-- Script -->
  <script>
    const buktiInput = document.getElementById("bukti");
    const preview = document.getElementById("preview");
    const previewImage = document.getElementById("previewImage");

    buktiInput.addEventListener("change", function () {
      const file = this.files[0];
      if (file) {
        const reader = new FileReader();
        reader.onload = function (e) {
          previewImage.src = e.target.result;
          preview.classList.remove("hidden");
        };
        reader.readAsDataURL(file);
      }
    });

    function simpanPembayaran() {
      const metode = document.getElementById("metode").value;
      const bukti = document.getElementById("bukti").files[0];

      if (!metode || !bukti) {
        alert("Lengkapi semua data pembayaran.");
        return false;
      }

      const reader = new FileReader();
      reader.onload = function (e) {
        const pembayaran = {
          metode: metode,
          buktiBase64: e.target.result,
          waktu: new Date().toISOString()
        };
        localStorage.setItem("pembayaranHexahustle", JSON.stringify(pembayaran));
        alert("Pembayaran disimpan di perangkat!");
        window.location.href = "beranda.html"; // Ganti dengan halaman beranda atau sukses
      };
      reader.readAsDataURL(bukti);

      return false; // prevent form submit
    }
  </script>
